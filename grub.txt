grub 添加windows启动项
vim /etc/grub.d/40_custom
menuentry "Windows 10/11" {
    insmod part_gpt
    insmod fat
    insmod search_fs_uuid
    insmod chain
    search --fs-uuid --set=root YOUR-ESP-UUID
    chainloader /EFI/Microsoft/Boot/bootmgfw.efi
}
