grub 添加windows启动项
vim /etc/grub.d/40_custom
menuentry "Windows 10/11" {
    insmod part_gpt
    insmod fat
    insmod search_fs_uuid
    insmod chain
    search --fs-uuid --set=root YOUR-ESP-UUID
    chainloader /EFI/Microsoft/Boot/bootmgfw.efi
}

grub命令行启动linux 
1、找到linux 内核文件和initramfs 镜像
		ls
2、告诉grub 它们在哪里
		set root=UUID=F750-9CBA
3、告诉内核根文件系统在哪里
		linux /vmlinuz-linux-lts root=UUID=

4、启动
		initrd /initramfs-linux-lts.img
		boot

grub重新生成
rm -r efi grub
grub-install --efi-directory=/boot 
grub-mkconfig -o /boot/grub/grub.cfg
